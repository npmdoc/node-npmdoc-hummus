    PDFRStreamForFile.prototype.read = function (inAmount)
{
    var buffer = new Buffer(inAmount*2);
    var bytesRead = fs.readSync(this.rs, buffer, 0, inAmount,this.rposition);
    var arr = [];

    for(var i=0;i<bytesRead;++i)
        arr.push(buffer[i]);
    this.rposition+=bytesRead;
    return arr;
